## Goal

Make `/curator/topology` render the **assembled** diagram — driven by `loadSystemTopology`
(C1/T1) → the T7 step engine → T3's `resolveSlots` → T4's per-shape template → T2's parts /
T5's mobile sheet — **instead of** the `@xyflow/react` + `@dagrejs/dagre` dot-canvas, while:

1. Keeping `ScenarioBar`, the kept Meter/reading block, and the captured/missing footer.
2. **KEEPING tap-any-shown-part-to-inspect** (Brandon override): the current step drives the

   `TopologyDetailPanel`. The selection mechanism is **adapted**, not deleted.
3. Adding a **whole-system escape control** that drops the tech into the **existing full
   faded-system view** (Brandon override — not a v1 placeholder).
4. Owning the **empty/null render states**: loader-null preserves not-found; a topology with
   zero implicated steps shows an honest "no test plan captured for this code yet" state — never
   a crash.
5. The mobile Meter is **tap-to-toggle** (peek ↔ expanded) via T5's `MeterSheet` — not a free-drag sheet.

T6 writes **zero** drawing logic. It is assembly-of-assemblies: it composes the upstream tracks'
pure outputs into the screen. No AI, no network, no per-case/per-system branching in the render path.

## Scalability bar — how T6 holds it

T6 contains **zero** system-specific or per-case branching. The screen calls
`buildStepSequence → resolveSlots → templateForShape` and renders whatever `ResolvedScene` it gets;
an unseen `kind`/role/`observationMethod`/`StepShape` is resolved by the upstream tracks' generic
fallbacks (T2 fallback part, T3 fallback shape, T4 fallback template) — T6 never inspects the
vocabulary. The zero-step state is the universal "needs field check" path, not an error. Adding a
new system/symptom/make is data-only: the same three calls render the new case with no T6 edit.
Tap-to-inspect routes by element id, never by part kind. Generality is exercised by the smoke test
across the fuel scene **and** a synthetic zero-step + a synthetic non-fuel `ResolvedScene` (proving
T6 renders any shape the engine emits, not just fuel).

/**
 * The render pipeline, as a pure function (no React, no DOM): the current
 * symptom's ordered steps (T7) → the current step → the resolved scene (T3) →
 * the per-shape template (T4). Returns `empty` when nothing is implicated — the
 * honest "needs field check" degrade, never a throw. No per-case/per-system
 * branching: it never inspects kind/role/observationMethod itself.
 */
export type StepView =
  | { kind: 'scene'; scene: ResolvedScene; Template: StepTemplate }
  | { kind: 'empty' }

export function assembleStepView(
  topology: SystemTopology,
  activeScenario: TopologyScenario | null,
): StepView {
  const steps = buildStepSequence(topology)
  const state = stepReducerInit(steps)
  const step = selectCurrentStep(state)
  if (step === null) return { kind: 'empty' }
  const scene = resolveSlots(topology, step, activeScenario)
  const Template = templateForShape(scene.shape)
  return { kind: 'scene', scene, Template }
}

**Step 2.3 — Minimal implementation: rewrite the component body.**

Replace the `TopologyDiagnostic` function (everything from `export function TopologyDiagnostic(`
through its closing `}`) in `components/screens/topology-diagnostic.tsx` with:
```tsx
type Props = {
  topology: SystemTopology
  layout: TopologyLayout
  vehicleName: string
  sessionId: string
}

type SelectionState =
  | { kind: 'empty' }
  | { kind: 'component'; id: string }
  | { kind: 'pin'; id: string }
  | { kind: 'connection'; id: string }

/**
 * Owns scenario state + free part selection (tap-to-inspect, KEPT per Brandon)
 * + the silent current-step (T7 reducer) + the whole-system escape. The current
 * step drives the default assembled view; the tech can tap any shown element to
 * pull its detail. No "step N of M" is ever shown — the index is internal.
 */
export function TopologyDiagnostic({
  topology,
  layout,
  vehicleName,
  sessionId,
}: Props) {
  const [selection, setSelection] = useState<SelectionState>({ kind: 'empty' })
  const [showWholeSystem, setShowWholeSystem] = useState(false)
  const [activeScenarioSlug, setActiveScenarioSlug] = useState<string | null>(
    () => defaultScenarioSlug(topology.scenarios, topology.lastScenarioSlug),
  )

  // T7 step engine: ordered steps + the silent current-step reducer. The index
  // is computed here and never surfaced as a counter.
  const steps = useMemo(() => buildStepSequence(topology), [topology])
  const [stepState, dispatchStep] = useReducer(
    stepReducer,
    steps,
    stepReducerInit,
  )

  const componentById = useMemo(
    () => new Map(topology.components.map((c) => [c.id, c])),
    [topology],
  )
  const pinById = useMemo(() => {
    const map = new Map<
      string,
      {
        pin: import('@/lib/diagnostics/load-system-topology').TopologyPin
        component: import('@/lib/diagnostics/load-system-topology').TopologyComponent
      }
    >()
    for (const c of topology.components) {
      for (const p of c.pins) map.set(p.id, { pin: p, component: c })
    }
    return map
  }, [topology])
  const connectionById = useMemo(
    () => new Map(topology.connections.map((c) => [c.id, c])),
    [topology],
  )
  const activeScenario = useMemo(
    () => topology.scenarios.find((s) => s.slug === activeScenarioSlug) ?? null,
    [topology, activeScenarioSlug],
  )

  // The assembled view for the current step (T3 + T4). Recomputed on step or
  // scenario change. selectCurrentStep keeps the reducer the single source.
  const stepView = useMemo(() => {
    const step = selectCurrentStep(stepState)
    if (step === null) return { kind: 'empty' as const }
    const scene = resolveSlots(topology, step, activeScenario)
    return { kind: 'scene' as const, scene, Template: templateForShape(scene.shape) }
  }, [stepState, topology, activeScenario])

  const panelSelection: TopologySelection = useMemo(() => {
    if (selection.kind === 'empty') return { kind: 'empty' }
    if (selection.kind === 'component') {
      const c = componentById.get(selection.id)
      return c ? { kind: 'component', component: c } : { kind: 'empty' }
    }
    if (selection.kind === 'connection') {
      const conn = connectionById.get(selection.id)
      return conn
        ? {
            kind: 'connection',
            connection: conn,
            fromComponent: componentById.get(conn.fromComponentId) ?? null,
            toComponent: componentById.get(conn.toComponentId) ?? null,
          }
        : { kind: 'empty' }
    }
    if (selection.kind === 'pin') {
      const entry = pinById.get(selection.id)
      return entry
        ? {
            kind: 'pin',
            pin: entry.pin,
            component: entry.component,
            scenario: activeScenario,
          }
        : { kind: 'empty' }
    }
    return { kind: 'empty' }
  }, [selection, componentById, connectionById, pinById, activeScenario])

  // Free selection KEPT: tapping any shown element resolves it to the right
  // selection kind by looking it up in the loaded graph (component first, then
  // pin). Routes by id, never by part kind — scalability-bar safe.
  const handleInspect = useCallback(
    (partId: string) => {
      if (componentById.has(partId)) setSelection({ kind: 'component', id: partId })
      else if (pinById.has(partId)) setSelection({ kind: 'pin', id: partId })
      else if (connectionById.has(partId)) setSelection({ kind: 'connection', id: partId })
    },
    [componentById, pinById, connectionById],
  )

  const diagramSelection: TopologySelectionState = useMemo(() => {
    if (selection.kind === 'empty') return { kind: 'empty', activeScenarioSlug }
    return { kind: selection.kind, id: selection.id, activeScenarioSlug }
  }, [selection, activeScenarioSlug])

  const persistScenario = useCallback(
    (slug: string) => {
      void fetch(`/api/sessions/${sessionId}/scenario`, {
        method: 'POST',
        headers: { 'content-type': 'application/json' },
        body: JSON.stringify({ slug }),
      }).catch(() => {
        // Fire-and-forget: a failed persistence write must not block the UI.
      })
    },
    [sessionId],
  )

  const handleScenarioChange = useCallback(
    (slug: string) => {
      setActiveScenarioSlug(slug)
      persistScenario(slug)
    },
    [persistScenario],
  )

  return (
    <div className="topo">
      <header className="topo__header">
        <Link
          href="/today"
          style={{
            fontFamily: 'var(--vt-font-mono)',
            fontSize: 11,
            letterSpacing: '0.08em',
            color: 'var(--vt-fg-3)',
            textDecoration: 'none',
          }}
        >
          ← Sessions
        </Link>
        <div className="topo__eyebrow">
          Electrical topology · diagnostic-complete from theory
        </div>
        <h1 className="topo__title">{formatSymptomTitle(topology.symptom.slug)}</h1>
        <div className="topo__vehicle">
          {vehicleName} · {topology.platform.name}
        </div>
      </header>

      {topology.scenarios.length > 0 && (
        <ScenarioBar
          scenarios={topology.scenarios}
          activeSlug={activeScenarioSlug}
          onScenarioChange={handleScenarioChange}
        />
      )}

      {activeScenario && (
        <div
          className={`topo__readout${activeScenario.kind === 'fault' ? ' is-fault' : ''}`}
        >
          Now showing · <b>{activeScenario.label}</b> — {activeScenario.sub}
        </div>
      )}

      <div className="topo__canvas-wrap">
        {showWholeSystem ? (
          // Brandon override: the whole-system button drops into the EXISTING
          // full faded-system view (xyflow), not a placeholder.
          <TopologyDiagram
            topology={topology}
            layout={layout}
            selection={diagramSelection}
            onSelectComponent={(id) => setSelection({ kind: 'component', id })}
            onSelectPin={(id) => setSelection({ kind: 'pin', id })}
            onClearSelection={() => setSelection({ kind: 'empty' })}
          />
        ) : stepView.kind === 'scene' ? (
          <div className="topo__assembled">
            <stepView.Template
              scene={stepView.scene}
              onInspect={handleInspect}
            />
          </div>
        ) : (
          <div className="topo__no-plan">
            <p className="topo__no-plan-title">No test plan captured for this code yet</p>
            <p className="topo__no-plan-sub">
              This case is online but its diagnostic steps aren’t authored. Use the
              whole-system view to inspect the parts, or check back as it fills in.
            </p>
          </div>
        )}

        <button
          type="button"
          className="topo__whole-system"
          aria-pressed={showWholeSystem}
          onClick={() => setShowWholeSystem((v) => !v)}
        >
          {showWholeSystem ? 'Back to step' : 'Whole system'}
        </button>

        <CapturedMissingFooter topology={topology} />
      </div>

      <TopologyDetailPanel
        selection={panelSelection}
        onSelectComponent={(id) => setSelection({ kind: 'component', id })}
        onSelectPin={(id) => setSelection({ kind: 'pin', id })}
        onClose={() => setSelection({ kind: 'empty' })}
        open={panelSelection.kind !== 'empty'}
      />
    </div>
  )
}
```

> `dispatchStep` is wired into the reducer now; advance/back controls are NOT added in v1 (the spec
