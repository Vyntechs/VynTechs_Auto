## Goal

Build the pure, deterministic, AI-free brain of the new diagram: given a loaded `SystemTopology`, one current `TopologyTestAction` (the step), and the active `TopologyScenario`, produce a `ResolvedScene` — the closed set of slots/parts/wires/overlay/gauge/verdict the templates render. The engine decides **what shows** (the leak-lock show-rule, keyed on the kind of test) and **where each part belongs** (slot resolution by data-derived role + a graph-walk circuit set). Zero system-specific or per-case branching: every decision is a pure function of the building-block vocabulary (`observationMethod`, `meterMode`, `stepKind`, `connectionKind`, `electricalRole`, `hasBranches`). Unseen values degrade to a generic fallback shape; null roles/pins degrade honestly (empty source/ground, never fabricated).

## Architecture

Three files under `lib/diagnostics/diagram/`, all pure (no React/DOM/network/AI imports):

- **`slot-interface.ts`** (C3, Wave 0 — type-only): the closed `SlotName` union, the 10-shape `StepShape` union, the `selectStepShape` signature, `SlotFill`/`ResolvedScene`/`AssembleScene`/`StepTemplate` types, the `ResolvedElement` enumerable type, and the typed per-shape `SHAPE_SLOT_RULES` table (required/optional/forbidden slots encoding the leak-lock).
- **`show-rule.ts`** (Wave 1): `selectStepShape(observationMethod, meterMode, stepKind, hasBranches) → StepShape` (maps ALL 9 `observationMethod` values + the electrical split by `meterMode` + locate via `stepKind` + fork via `hasBranches`; `waveform_capture` degrades to a neutral shape; unseen → generic fallback) and `computeShowBudget(shape) → { pinsAllowed }` (true ONLY for electrical shapes).
- **`slot-resolver.ts`** (Wave 1): `resolveSlots` / `assembleScene` (the `AssembleScene` impl) — graph-walks the circuit set from the focus component, assigns `device-under-test`/`source`/`ground`/`downstream-anchor` from data-derived roles, emits exactly one overlay + `activeWireIds`, builds the `detail` content payload from C1 prose, computes the single verdict signal (out-of-range → branch-fail → neutral), and populates `ResolvedScene.elements` for the leak test.

**Data realities that shape the design (verified against `.design-shots/scene-data.json`):**
- The real seeded scene has `electrical_role`, `from_pin_id`, `to_pin_id` ALL null on every connection, and ships ZERO `test_actions`. So: (a) the honest-degrade path (empty source/ground, no terminals) is the steady state and must be tested against a real-shaped fixture; (b) the electrical-probe slot wiring must be proven on a **synthetic** `TopologyTestAction` fixture.
- There is **no step→circuit-set FK**. The circuit set is derived by a bounded graph-walk over `connections` from the focus component (the step's owning component), per the scope card.

  switch (meterMode) {
    case 'ohms':
      return 'continuity-ground'
    case 'drop':
      return 'voltage-drop'
    case 'duty':
      return 'duty-pwm'
    case 'volts':
    case 'amps':
    case 'pressure': // a pressure meter on an electrical method still probes
    case 'pid':
    case null:
    default:
      return 'electrical-probe'
  }

export const selectStepShape = (
  observationMethod: string,
  meterMode: MeterMode | null,
  stepKind: string | null,
  hasBranches: boolean,
): StepShape => {
  // stepKind=locate/orient/find overrides any method — it is a placement step.
  if (stepKind !== null && LOCATE_STEP_KINDS.includes(stepKind)) {
    return 'locate'
  }

  // The base shape from the KIND OF TEST.
  let base: StepShape
  if (observationMethod === 'electrical_measurement_at_pin') {
    base = electricalSplit(meterMode)
  } else if (observationMethod === 'pressure_test_with_gauge') {
    base = 'pressure-flow'
  } else if (observationMethod === 'scan_tool_pid') {
    base = 'single-pid'
  } else if (LOOK_METHODS.includes(observationMethod)) {
    base = 'look-inspect'
  } else if (observationMethod === 'waveform_capture') {
    // No reference trace in v1 → degrade to a neutral single reading.
    base = 'single-pid'
  } else {
    // Generic fallback for any UNSEEN method — a neutral single reading.
    base = 'single-pid'
  }

  // A branchy NON-electrical reading is a fork (the electrical shapes already
  // own the source/DUT/ground arrangement; branches there route via the Meter).
  if (hasBranches && base !== 'electrical-probe' && base !== 'continuity-ground'
      && base !== 'voltage-drop' && base !== 'duty-pwm') {
    return 'fork'
  }

  return base
}

 * slot-resolver — the AssembleScene body (owner T3). Pure; NO React/DOM/

 * data-derived role, emits one overlay, computes the verdict signal, and
 * degrades honestly on null roles/pins. Zero per-system / per-case branching.

import type {
  SystemTopology,
  TopologyComponent,
  TopologyTestAction,
  TopologyScenario,

function findFocus(topology: SystemTopology, step: TopologyTestAction): TopologyComponent | null {
  return (
    topology.components.find((c) =>
      c.testActions.some((t) => t.slug === step.slug),
    ) ?? null
  )

/**
 * Bounded breadth-first walk of the connection graph from the focus component.
 * Returns the focus + every component within `depth` hops. No step->circuit FK
 * exists, so the circuit set IS this walk. Islands (no path to focus) excluded.
 */
export function walkCircuitSet(
  topology: SystemTopology,
  focusId: string,
  depth: number,
): TopologyComponent[] {
  const byId = new Map(topology.components.map((c) => [c.id, c]))
  if (!byId.has(focusId)) return []
  const visited = new Set<string>([focusId])
  let frontier = [focusId]
  for (let d = 0; d < depth; d++) {
    const next: string[] = []
    for (const conn of topology.connections) {
      if (frontier.includes(conn.fromComponentId) && !visited.has(conn.toComponentId)) {
        visited.add(conn.toComponentId); next.push(conn.toComponentId)
      }
      if (frontier.includes(conn.toComponentId) && !visited.has(conn.fromComponentId)) {
        visited.add(conn.fromComponentId); next.push(conn.fromComponentId)
      }
    }
    frontier = next
    if (frontier.length === 0) break
  }
  return [...visited].map((id) => byId.get(id)!).filter(Boolean)
}

/** Provenance by data, not a grade: amber needs-field-check from labelGap/
 *  electricalContract gaps; field-verified from sourceProvenance; else drafted. */
function provenanceOf(c: TopologyComponent): PartProvenance {
  if (c.sourceProvenance === 'FIELD-VERIFIED') return 'field-verified'
  return 'drafted'
}

    partId: c.id,
    kind: c.kind as PartKind,
    roleSpecial: null,
    tier,
    provenance: provenanceOf(c),
    terminals: [],
    selected,

}

  return {
    shape,

    activeWireIds: [],
    overlay: null,
    gaugeSpec: null,
    forkRoute: null,
    focus: { selectedPartId: focus?.id ?? '' },
    pinsAllowed: budget.pinsAllowed,
    elements,
  }
}

 *   isOutOfRange authoritative -> branch verdict==='fail' -> neutral default.

 * this; T3 never decides red twice.

export function computeVerdict(
  step: TopologyTestAction,
  scenario: TopologyScenario,
  focusPins: TopologyPin[],
): VerdictSignal {
  const outMap = (scenario as { isOutOfRange?: Record<string, boolean> }).isOutOfRange
  if (outMap) {
    for (const pin of focusPins) {
      if (outMap[pin.id] === true) return 'out-of-range'
    }
  }
  if (step.branches.some((b) => b.verdict === 'fail')) return 'branch-fail'

}
